const SHEET_NAME = 'Tabelle1'; function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const params = e.parameter; const action = params.action; const callback = params.callback || 'callback'; let result; if (action === 'read') { const data = sheet.getDataRange().getValues(); const rows = []; for (let i = 1; i < data.length; i++) { if (data[i][0]) { rows.push({ tag: String(data[i][0]), schicht: String(data[i][1]), platz1: String(data[i][2] || ''), platz2: String(data[i][3] || '') }); } } result = JSON.stringify({ ok: true, data: rows }); } else if (action === 'write') { const tag = params.tag; const schicht = params.schicht; const platz = parseInt(params.platz); const name = params.name || ''; const data = sheet.getDataRange().getValues(); let rowIndex = -1; for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === tag && String(data[i][1]) === schicht) { rowIndex = i + 1; break; } } if (rowIndex === -1) { const newRow = [tag, schicht, '', '', new Date().toISOString()]; newRow[platz + 1] = name; sheet.appendRow(newRow); } else { sheet.getRange(rowIndex, platz + 1).setValue(name); sheet.getRange(rowIndex, 5).setValue(new Date().toISOString()); } result = JSON.stringify({ ok: true }); } else { result = JSON.stringify({ ok: false, error: 'Unknown action' }); } return ContentService .createTextOutput(callback + '(' + result + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); }
